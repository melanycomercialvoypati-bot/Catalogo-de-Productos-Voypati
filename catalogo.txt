        let currentProducts = [...products];
        let activeCategory = 'Todos';
        let activeSort = { sort: 'price', order: 'asc' };

        const renderProducts = (productsToRender) => {
            productGrid.innerHTML = '';
            if (productsToRender.length === 0) {
                productGrid.innerHTML = `<p class="col-span-full text-center text-orange-500">No se encontraron productos que coincidan con su selecci√≥n.</p>`;

        const setupCategories = () => {
            const categories = ['Todos', ...new Set(products.map(p => p.category))];
            categoryFiltersContainer.innerHTML = categories.map(category => `
                <button class="nav-button text-sm px-4 py-2 bg-white border border-orange-300 rounded-full ${category === 'Todos' ? 'active' : ''}" data-category="${category}">
                    ${category}
                </button>
            `).join('');
        };
        
        const applyFiltersAndSort = () => {
            let filtered = activeCategory === 'Todos' ? [...products] : products.filter(p => p.category === activeCategory);
            
            filtered.sort((a, b) => {
                const order = activeSort.order === 'asc' ? 1 : -1;
                if (activeSort.sort === 'price') {
                    return (a.price - b.price) * order;
                }
                if (activeSort.sort === 'name') {
                    return a.name.localeCompare(b.name) * order;
                }
                return 0;
            });
            
            currentProducts = filtered;
            renderProducts(currentProducts);
        };
        
        categoryFiltersContainer.addEventListener('click', (e) => {
            if (e.target.tagName === 'BUTTON') {
                activeCategory = e.target.dataset.category;
                document.querySelectorAll('#category-filters .nav-button').forEach(btn => btn.classList.remove('active'));
                e.target.classList.add('active');
                applyFiltersAndSort();
            }
        });

        sortControlsContainer.addEventListener('click', (e) => {

                e.target.classList.add('active');

            }
        });

            setupCategories();
